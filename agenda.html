/* global luxon, jQuery */ const $ = jQuery; // ✅ CONFIGURACIÓN PARA MAIK SPORT const AGENDA_URL = "https://pltvhd.com/diaries.json"; const IMG_BASE = "https://pltvhd.com"; $(document).ready(function () { // 1. Carga inicial obtenerAgenda(); // 2. Acordeón: Al tocar el partido, abre los canales $(document).on("click", ".toggle-submenu", function (e) { // Si toca el link directo del canal, que abra el video y no cierre el menú if ($(e.target).closest('a').length) return; const $submenu = $(this).find("ul"); $(".agenda-list ul").not($submenu).slideUp(); // Cierra otros $submenu.slideToggle(); // Abre este }); // 3. Auto-refrescar cada 2 minutos setInterval(obtenerAgenda, 120000); }); async function obtenerAgenda() { const menuElement = document.getElementById("menu"); const titleElement = document.getElementById("title-agenda"); // Si el ID 'menu' no existe en el HTML, el código se detiene aquí if (!menuElement) return; try { const response = await fetch(AGENDA_URL, { cache: "no-store" }); const result = await response.json(); if (!result || !result.data) return; let html = ""; result.data.forEach((evento) => { const attr = evento.attributes; const embeds = attr.embeds?.data || []; // LÓGICA DE ICONOS (Champions, NBA, etc) let imageUrl = `${IMG_BASE}/uploads/sin_imagen_d36205f0e8.png`; const flagPath = attr.country?.data?.attributes?.image?.data?.attributes?.url; if (flagPath) { imageUrl = flagPath.startsWith('http') ? flagPath : `${IMG_BASE}${flagPath}`; } // HORA LOCAL const hora = attr.diary_hour ? attr.diary_hour.substring(0, 5) : "--:--"; html += `
  • ${hora} ${attr.diary_description}
      ${embeds.map(emb => { let finalHref = emb.attributes.embed_iframe; // Si el link es relativo lo arreglamos para que apunte a tu carpeta if (finalHref.startsWith('/embed/')) finalHref = "." + finalHref; return `
    • ▶ ${emb.attributes.embed_name}
      • `; }).join('')}
    • `; }); menuElement.innerHTML = html; if (titleElement) titleElement.innerHTML = "Agenda Deportiva"; } catch (error) { console.error("❌ Fallo en la conexión con el servidor de señales"); } }
