/* global luxon, jQuery */ const $ = jQuery; const AGENDA_URL = "https://pltvhd.com/diaries.json"; const IMG_BASE = "https://pltvhd.com"; $(document).ready(function () { // Carga la agenda apenas abre la página obtenerAgenda(); // Abre y cierra los canales al tocar el partido $(document).on("click", ".toggle-submenu", function (e) { if ($(e.target).closest('a').length) return; $(this).find("ul").slideToggle(); $(this).toggleClass("active"); }); // Actualiza cada 2 minutos solito setInterval(obtenerAgenda, 120000); }); async function obtenerAgenda() { const menuElement = document.getElementById("menu"); const titleElement = document.getElementById("title-agenda"); // Si no encuentra el ID 'menu', no hace nada (evita que la web se rompa) if (!menuElement) return; try { const response = await fetch(AGENDA_URL); const result = await response.json(); if (!result.data) return; let html = ""; result.data.forEach((evento) => { const attr = evento.attributes; const embeds = attr.embeds?.data || []; // Lógica para que los logos de Champions, NBA, etc., carguen bien let imageUrl = `${IMG_BASE}/uploads/sin_imagen_d36205f0e8.png`; const flagPath = attr.country?.data?.attributes?.image?.data?.attributes?.url; if (flagPath) { imageUrl = flagPath.startsWith('http') ? flagPath : `${IMG_BASE}${flagPath}`; } // Hora en formato HH:mm const hora = attr.diary_hour ? attr.diary_hour.substring(0, 5) : "--:--"; html += `
  • ${hora} ${attr.diary_description}
      ${embeds.map(emb => { // Ajuste automático de la ruta del reproductor let link = emb.attributes.embed_iframe; if (link.startsWith('/embed/')) link = "." + link; return `
    • ▶ ${emb.attributes.embed_name}
      • `; }).join('')}
    • `; }); menuElement.innerHTML = html; if (titleElement) titleElement.innerHTML = "Agenda - Partidos de Hoy"; } catch (error) { console.error("Error cargando la agenda"); } }
